declare(strict_types=1);

namespace BaksDev\Products\Supply\Controller\Admin;


use BaksDev\Core\Controller\AbstractController;
use BaksDev\Core\Listeners\Event\Security\RoleSecurity;
use BaksDev\Delivery\Entity\Delivery;
use BaksDev\Delivery\Entity\Event\DeliveryEvent;
use BaksDev\Delivery\UseCase\Admin\NewEdit\DeliveryDTO;
use BaksDev\Delivery\UseCase\Admin\NewEdit\DeliveryForm;
use BaksDev\Delivery\UseCase\Admin\NewEdit\DeliveryHandler;
use BaksDev\Products\Supply\Entity\Event\ProductSupplyEvent;
use BaksDev\Products\Supply\Entity\ProductSupply;
use BaksDev\Products\Supply\UseCase\Admin\NewEdit\NewEditProductSupplyDTO;
use BaksDev\Products\Supply\UseCase\Admin\NewEdit\NewEditProductSupplyForm;
use BaksDev\Products\Supply\UseCase\Admin\NewEdit\NewEditProductSupplyHandler;
use Symfony\Bridge\Doctrine\Attribute\MapEntity;
use Symfony\Component\HttpFoundation\Request;
use Symfony\Component\HttpFoundation\Response;
use Symfony\Component\Routing\Attribute\Route;
use Symfony\Component\HttpKernel\Attribute\AsController;

#[AsController]
#[RoleSecurity('ROLE_PRODUCT_SUPPLY_EDIT')]
final class EditController extends AbstractController
{
    public const string NAME = 'admin.newedit.edit';

    #[Route('/admin/product/supply/edit/{id}', name: self::NAME, methods: ['GET', 'POST'])]
    public function edit(
        Request $request,
        #[MapEntity] ProductSupplyEvent $ProductSupplyEvent,
        NewEditProductSupplyHandler $NewEditProductSupplyHandler,
    ): Response
    {
        $NewEditProductSupplyDTO = new NewEditProductSupplyDTO();
        $ProductSupplyEvent->getDto($NewEditProductSupplyDTO);

        // Форма
        $NewEditProductSupplyForm = $this
            ->createForm(
                type: NewEditProductSupplyForm::class,
                data: $NewEditProductSupplyDTO,
                options: [
                    'action' => $this->generateUrl(
                        route: 'product-supply:admin.newedit.edit',
                        parameters: ['id' => $NewEditProductSupplyDTO->getEvent()],
                    ),
                ])
            ->handleRequest($request);

        if(
            $NewEditProductSupplyForm->isSubmitted() &&
            $NewEditProductSupplyForm->isValid() &&
            $NewEditProductSupplyForm->has('product_supply_add')
        )
        {
            $handle = $NewEditProductSupplyHandler->handle($NewEditProductSupplyDTO);

            $flash = $this->addFlash
            (
                'page.edit',
                $handle instanceof ProductSupply ? 'success.edit' : 'danger.edit',
                'product.supply.admin',
                $handle,
            );

            return $flash ?: $this->redirectToRoute('products-supply:admin.supply.index');
        }

        return $this->render(['form' => $NewEditProductSupplyForm->createView()]);
    }
}
